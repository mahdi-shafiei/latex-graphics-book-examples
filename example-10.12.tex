\pgfplotsset{scale only axis, width=3cm, height=5cm,
  every axis plot/.style={thick,mark size=3pt},
  A/.style={blue,mark=*,
    every mark/.append style={fill=.!20}},
  B/.style={orange,mark=star,dotted,
    every mark/.append style={solid}},
  C/.style={mark=halfcircle*,mark color=yellow,
    every mark/.append style={rotate=45}}}
\begin{tikzpicture}
  \begin{axis}[legend entries={A,B,C}]
    \addplot[A] table[y=A]{data.dat};
    \addplot[B] table[y=B]{data.dat};
    \addplot[C] table[y=C]{data.dat};
  \end{axis}
\end{tikzpicture}
