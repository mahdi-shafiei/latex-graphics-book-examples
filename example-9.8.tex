\usepackage{parskip}

%% *** Enable PGFPLOTS (automatically enables TikZ) ***
\usepackage{pgfplots}

%% Prevent some PGFPLOTS messages (not critical)
\pgfplotsset{compat=1.18,compat/show suggested version=false}

%% *** PGFPLOTS library ***
\usepgfplotslibrary{fillbetween}

\begin{document}

%% *** START OF EXAMPLE CODE ***
\begin{tikzpicture}
  \begin{axis}[scale only axis, width=6cm, height=3cm,
      xticklabel style={/pgf/number format/1000 sep=},font=\small,
      ymin=0, legend pos=south east,
      legend entries={max,min,difference}
    ]
    \addplot[red,name path=max]  table[y=max af]{res.dat};
    \addplot[blue,name path=min] table[y=min af]{res.dat};
    \addplot[fill=yellow] fill between [of=max and min];
  \end{axis}
\end{tikzpicture}
%% *** END OF EXAMPLE CODE ***

\end{document}
