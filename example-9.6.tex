  \begin{axis}[

      legend entries={Alpha,Beta,Gamma,Delta},
      legend style={ultra thin,fill opacity=0.7,

    \addplot[blue] table[y=A]{\data};
    \addplot[red,mark=x] table[y=B]{\data};
    \addplot[orange,mark=+] table[y=C]{\data};
    \addplot[teal,mark=star] table[y=D]{\data};
  \end{axis}
