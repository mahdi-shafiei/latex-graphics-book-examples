\begin{tikzpicture}[>=Triangle,
    circ/.style={circle, draw},
    connect/.style={auto, sloped, inner sep=0.1em}]
  \node[circ] at (0,0) (A) {A};
  \node[circ] at (-1,-1.5) (B) {B}
    edge[->] node[connect]{flow} (A) ;
  \node[circ] at (0,-3) (C) {C}
    edge[->] node[connect]{flow} (B);
  \path (A) edge[->,bend left] node[connect]{flow} (C);
\end{tikzpicture}
