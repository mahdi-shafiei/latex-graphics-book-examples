\pgfplotsset{table/col sep=comma}
\begin{tikzpicture}
  \begin{axis}[scale only axis,
      width=3cm, height=3cm, xtick distance=1,
      only marks, scatter,
      scatter src=explicit symbolic,
      title=Scatter Plot,
      legend entries={Class a,Class b}]
    \addplot[scatter/classes={
        a={blue,mark=*},
        b={red,mark=square*} % No comma
      }] table[meta=class]{points.csv};
  \end{axis}
\end{tikzpicture}
